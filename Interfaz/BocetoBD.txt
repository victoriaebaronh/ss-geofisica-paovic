    correspondencia_paqueteria BOOLEAN,

    mantenimiento_planta_luz BOOLEAN,
    mantenimiento_otro BOOLEAN,
    reproduccion_fotocopiado BOOLEAN,
    reproduccion_engargolado BOOLEAN,
    reproduccion_otro BOOLEAN,
    transporte_local BOOLEAN,
    transporte_foraneo BOOLEAN,
    transporte_pasajero BOOLEAN,
    transporte_carga BOOLEAN,
    albanileria BOOLEAN,
    carpinteria BOOLEAN,
    herreria BOOLEAN,

    electricidad BOOLEAN,

    vigilancia_eventos_especiales BOOLEAN,
